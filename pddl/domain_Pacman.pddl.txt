(define (domain domain_Pacman)


	(:predicates      (pacmanAt ?p) ;; true iff Pacman at position p
				(ghostAt ?p)	;; true iff opposite ghost at position p
				(foodAt ?p)	;; true iff food at position p
				(capsuleAt ?p)	;; true iff power capsule at position p
				(adjacent ?p1 ?p2);;true iff p1osition1 and position2 are adjacent
	;;
	;;被吃掉后怎么办？
	(:action move : parameters (?p1 ?p2) ;;move from p1 to p2
				:precondition (and   (pacmanAt ?p)
								(adjacent ?p1 ?p2)
							)
				:effect (and      (pacmanAt p2)
							(not (pacmanAt p1))
							(not (foodAt p2))
							(not capsuleAt p2)
							(when (powered) (not ghostAt p2))					
	)

)
